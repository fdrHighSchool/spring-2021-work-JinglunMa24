private String[] mixedWords(String[] words) {
  String[] output = new String[words.length];
  for (int i = 0; i < words.length; i += 2) {
    output[i] = recombine(words[i], words[i + 1]);
    output[i + 1] = recombine(words[i + 1], words[i]);
  } //end for loop
  return output;
} //end mixedWords()
